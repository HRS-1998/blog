排序方法总结 ​

冒泡排序 ​

冒泡排序（Bubble Sort）基本思想： 经过多次迭代，通过相邻元素之间的比较与交换，使值较小的元素逐步从后面移到前面，值较大的元素从前面移到>后面

js
// 冒泡排序
function bubbleSort(arr) {
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
      }
    }
  }
  return arr;
}

// 优化，增加标记，如果某一次循环没有发生交换，则说明数组已经有序，可以提前结束循环。
function bubbleSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n; i++) {
    let swapped = false; // 标志位，用于检测是否发生交换
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]; // 交换元素
        swapped = true; // 设置标志位为 true，表示发生了交换
      }
    }
    if (!swapped) {
      break; // 如果没有发生交换，提前终止排序
    }
  }
  return arr;
}

选择排序 ​

选择排序（Selection Sort）基本思想： 将数组分为两个区间：左侧为已排序区间，右侧为未排序区间。每趟从未排序区间中选择一> 个值最小的元素，放到已排序区间的末尾，从而将该元素划分到已排序区间。

js
// 选择排序
function selectionSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    let minIndex = i;
    for (let j = i + 1; j < n; j++) {
      if (arr[j] < arr[minIndex]) {
        minIndex = j; // 找到最小元素的索引
      }
    }
    if (minIndex !== i) {
      [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]]; // 交换元素
    }
  }
  return arr;
}

插入排序 ​

插入排序（Insertion Sort）基本思想： 将数组分为有序区间和无序区间，每次从未排序区间中选择一个元素，将其插入到已排序区间的适当位置，从而将未排序区间中的元素逐步归并到已排序区间中。

js
// 插入排序
function insertionSort(arr) {
  for (let i = 1; i < arr.length; i++) {
    let current = arr[i];
    let j = i - 1;
    // 将当前元素插入到已排序部分的正确位置
    while (j >= 0 && arr[j] > current) {
      arr[j + 1] = arr[j];
      j--;
    }
    arr[j + 1] = current; // 插入当前元素
  }
  return arr;
}

希尔排序 ​

希尔排序（Shell Sort）基本思想： 希尔排序是插入排序的改进版，它通过不断缩小增量（gap）来优化插入排序，从而达到更快的排序速度。 将整个数组切按照一定的间隔取值划分为若干个子数组，每个子数组分别进行插入排序。然后逐渐缩小间隔进行下一轮划分子数组和对子数组进行插入排序。直至最后一轮排序间隔为 1，对整个数组进行插入排序。

js
function shellSort(arr) {
  const n = arr.length;

  // 动态生成Sedgewick序列
  const sedgewick = () => {
    const gaps = [];
    let k = 0;
    while (true) {
      const gap = Math.ceil(9 * Math.pow(4, k) - 9 * Math.pow(2, k) + 1);
      if (gap > n / 2) break; // 超过半数即停止
      gaps.unshift(gap); // 保证降序
      k++;
    }
    return gaps.length ? gaps : [1]; // 最小保障gap=1
  };

  for (let gap of sedgewick()) {
    for (let i = gap; i < n; i++) {
      const current = arr[i];
      let j = i;
      while (j >= gap && arr[j - gap] > current) {
        arr[j] = arr[j - gap];
        j -= gap;
      }
      arr[j] = current;
    }
  }
  return arr;
}

归并排序 ​

归并排序（Merge Sort）基本思想： 归并排序是分治算法的一种，它将数组分为两个子数组，分别进行排序，然后将两个子数组归并成一个有序数组。

js
// 基础实现
// 归并排序
function mergeSort(arr) {
  if (arr.length <= 1) return arr;

  // 分割点计算
  const mid = Math.floor(arr.length / 2);

  // 递归分割
  const left = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));

  // 合并有序数组
  return merge(left, right);
}

// 合并两个有序数组
function merge(left, right) {
  let result = [];
  let i = 0,
    j = 0;

  while (i < left.length && j < right.length) {
    result.push(left[i] < right[j] ? left[i++] : right[j++]);
  }

  // 拼接剩余元素
  return result.concat(i < left.length ? left.slice(i) : right.slice(j));
}

// 优化实现
function mergeSort(arr) {
  const n = arr.length;
  let temp = new Array(n); // 预分配临时空间

  for (let size = 1; size < n; size *= 2) {
    for (let leftStart = 0; leftStart < n; leftStart += 2 * size) {
      const mid = Math.min(leftStart + size, n);
      const rightEnd = Math.min(leftStart + 2 * size, n);
      merge(arr, leftStart, mid, rightEnd, temp);
    }
  }
  return arr;
}

function merge(arr, left, mid, right, temp) {
  let i = left,
    j = mid,
    k = left;

  while (i < mid && j < right) {
    temp[k++] = arr[i] < arr[j] ? arr[i++] : arr[j++];
  }

  // 复制剩余元素
  while (i < mid) temp[k++] = arr[i++];
  while (j < right) temp[k++] = arr[j++];

  // 回写数据
  for (let p = left; p < right; p++) {
    arr[p] = temp[p];
  }
}

快速排序 ​

快速排序（Quick Sort）基本思想： 快速排序是分治算法的一种，它通过选择一个基准元素，将数组分为两个子数组，使得子数组 1 中的元素都小于等于基准元素，子数组 2 中的元素都大于等于基准元素。然后分别对两个子数组进行快速排序。

js
//基础实现
// 快速排序基础版
function quickSort(arr) {
  if (arr.length <= 1) return arr;

  const pivot = arr[arr.length - 1]; // 选取最后元素为基准
  const left = [];
  const right = [];

  for (const item of arr.slice(0, -1)) {
    (item < pivot ? left : right).push(item);
  }

  return [...quickSort(left), pivot, ...quickSort(right)];
}

优化版本

js
// 快速排序优化版
function quickSort(arr, left = 0, right = arr.length - 1) {
  if (left >= right) return;

  // 三数取中法选择基准
  const pivotIndex = medianOfThree(arr, left, right);
  const partitionIndex = hoarePartition(arr, left, right, pivotIndex);

  // 尾递归优化
  quickSort(arr, left, partitionIndex - 1);
  quickSort(arr, partitionIndex + 1, right);

  return arr;
}

// Hoare分区方案
function hoarePartition(arr, left, right, pivotIndex) {
  const pivotValue = arr[pivotIndex];
  [arr[pivotIndex], arr[right]] = [arr[right], arr[pivotIndex]]; // 基准移到最后
  let storeIndex = left;

  for (let i = left; i < right; i++) {
    if (arr[i] <= pivotValue) {
      [arr[storeIndex], arr[i]] = [arr[i], arr[storeIndex]];
      storeIndex++;
    }
  }
  [arr[storeIndex], arr[right]] = [arr[right], arr[storeIndex]];
  return storeIndex;
}

// 三数取中法
function medianOfThree(arr, a, b, c = Math.floor((a + b) / 2)) {
  const values = [
    [arr[a], a],
    [arr[b], b],
    [arr[c], c],
  ];
  values.sort((x, y) => x[0] - y[0]);
  return values[1][1]; // 返回中间值的索引
}

堆排序 ​

堆排序（Heap Sort）基本思想： 堆排序是选择排序的一种，它通过构建一个堆来选择排序。堆是一个完全二叉树，它的每个节点都大于等于它的子节点。 堆排序的过程是：

  1. 将数组构建成一个大顶堆。
  2. 将堆顶元素与最后一个元素交换，然后将堆顶元素“下沉”到正确的位置。
  3. 重复步骤 2，直到堆中只有一个元素。
  4. 最后，堆中的元素就是排序后的数组。
  5. 堆排序的时间复杂度为 O(nlogn)，空间复杂度为 O(1)。
  6. 堆排序的稳定性为不稳定。
js
// 堆排序 原地排序
function heapSort(arr) {
  const n = arr.length;

  // 构建最大堆（从最后一个非叶子节点开始）
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i);
  }

  // 逐个提取最大值到数组末尾
  for (let size = n - 1; size > 0; size--) {
    [arr[0], arr[size]] = [arr[size], arr[0]]; // 交换堆顶与末尾元素
    heapify(arr, size, 0); // 调整剩余元素的堆结构
  }
  return arr;
}

// 堆调整函数（下沉操作）
function heapify(arr, size, root) {
  let largest = root;
  const left = 2 * root + 1;
  const right = 2 * root + 2;

  // 找出根节点、左右子节点中的最大值
  if (left < size && arr[left] > arr[largest]) largest = left;
  if (right < size && arr[right] > arr[largest]) largest = right;

  // 若最大值不是根节点，则交换并递归调整
  if (largest !== root) {
    [arr[root], arr[largest]] = [arr[largest], arr[root]];
    heapify(arr, size, largest);
  }
}

优化版本(迭代式堆调整)

js
function heapSort(arr) {
  const n = arr.length;

  // 使用非递归方式构建初始堆
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapifyIterative(arr, n, i);
  }

  // 优化交换逻辑（减少临时变量）
  for (let size = n - 1; size > 0; size--) {
    [arr[0], arr[size]] = [arr[size], arr[0]];
    heapifyIterative(arr, size, 0);
  }
  return arr;
}

// 迭代式堆调整
function heapifyIterative(arr, size, root) {
  let current = root;
  while (true) {
    const left = 2 * current + 1;
    const right = 2 * current + 2;
    let largest = current;

    if (left < size && arr[left] > arr[largest]) largest = left;
    if (right < size && arr[right] > arr[largest]) largest = right;

    if (largest === current) break;

    [arr[current], arr[largest]] = [arr[largest], arr[current]];
    current = largest;
  }
}

计数排序 ​

计数排序（Counting Sort）基本思想： 计数排序是一种基于计数的排序算法，它通过统计每个元素出现的次数来确定元素的排序顺序。

js
// 计数排序基础版（仅支持非负整数）
function countingSort(arr) {
  const max = Math.max(...arr);
  const countArray = new Array(max + 1).fill(0);

  // 统计元素出现次数
  for (const num of arr) {
    countArray[num]++;
  }

  // 生成排序结果
  let sortedIndex = 0;
  for (let num = 0; num < countArray.length; num++) {
    while (countArray[num] > 0) {
      arr[sortedIndex++] = num;
      countArray[num]--;
    }
  }
  return arr;
}

优化版本（支持负数 + 稳定排序）

js
// 计数排序优化版（支持所有整数）
function countingSort(arr) {
  const min = Math.min(...arr);
  const max = Math.max(...arr);
  const range = max - min + 1;
  const countArray = new Array(range).fill(0);

  // 统计偏移量后的元素出现次数
  for (const num of arr) {
    countArray[num - min]++;
  }

  // 计算累计分布（保证排序稳定性）
  for (let i = 1; i < range; i++) {
    countArray[i] += countArray[i - 1];
  }

  // 反向填充结果数组
  const output = new Array(arr.length);
  for (let i = arr.length - 1; i >= 0; i--) {
    const pos = arr[i] - min;
    output[countArray[pos] - 1] = arr[i];
    countArray[pos]--;
  }

  return output;
}

桶排序 ​

桶排序（Bucket Sort）基本思想： 桶排序是一种基于计数的排序算法，它通过将元素分配到桶中来确定元素的排序顺序。 chrome v8 Array.prototype.sort()在数据量大于 1 万时会使用桶排序与插入排序的混合策略

js
// 桶排序基础版（适合均匀分布数值）
function bucketSort(arr, bucketSize = 5) {
  const max = Math.max(...arr);
  const min = Math.min(...arr);
  const bucketCount = Math.floor((max - min) / bucketSize) + 1;
  const buckets = Array.from({ length: bucketCount }, () => []);

  // 分桶操作
  for (const num of arr) {
    const index = Math.floor((num - min) / bucketSize);
    buckets[index].push(num);
  }

  // 各桶内排序并合并
  return buckets.flatMap((bucket) => {
    return bucket.sort((a, b) => a - b);
  });
}
js
// 桶排序优化版（支持任意分布数据）
function bucketSort(arr, bucketSize = 5) {
  if (arr.length === 0) return arr;

  // 动态计算桶数量
  const max = Math.max(...arr);
  const min = Math.min(...arr);
  const range = max - min;
  bucketSize = Math.min(
    Math.max(1, Math.floor(range / Math.sqrt(arr.length))),
    bucketSize
  );

  // 初始化桶
  const bucketCount = Math.ceil(range / bucketSize) + 1;
  const buckets = Array.from({ length: bucketCount }, () => []);

  // 分桶策略优化（防止空桶）
  for (const num of arr) {
    const index = Math.min(
      Math.floor((num - min) / bucketSize),
      bucketCount - 1
    );
    insertSort(buckets[index], num); // 使用插入排序维护有序性
  }

  return buckets.flat();
}

// 带插入的排序方法（保持稳定性）
function insertSort(bucket, num) {
  let pos = bucket.length;
  while (pos > 0 && bucket[pos - 1] > num) {
    pos--;
  }
  bucket.splice(pos, 0, num);
}

基数排序 ​

基数排序（Radix Sort）基本思想： 基数排序它的基本思想是将整数按位数切割成不同的数字，然后从低位开始，依次到高位，逐位进行排序，从而达到排序的目的。

js
// 基数排序基础版（非负整数）
function radixSort(arr) {
  const max = Math.max(...arr);
  const maxDigit = String(max).length;
  let mod = 10,
    dev = 1;

  for (let i = 0; i < maxDigit; i++, dev *= 10, mod *= 10) {
    // 初始化10个桶（0-9）
    const buckets = Array.from({ length: 10 }, () => []);

    // 分配到桶中
    for (const num of arr) {
      const bucketIndex = Math.floor((num % mod) / dev);
      buckets[bucketIndex].push(num);
    }

    // 合并桶数据
    arr = [].concat(...buckets);
  }
  return arr;
}

优化版本

js
// 基数排序优化版（支持所有整数）
function radixSort(arr) {
  // 分离正负数处理
  const negatives = arr.filter((n) => n < 0).map((n) => -n);
  const positives = arr.filter((n) => n >= 0);

  function sort(nums) {
    const max = Math.max(...nums);
    const maxByte = Math.ceil(Math.log2(max + 1) / 8); // 计算最大字节数
    const buffer = new ArrayBuffer(4);
    const view = new DataView(buffer);

    for (let byte = 0; byte < maxByte; byte++) {
      const buckets = Array.from({ length: 256 }, () => []); // 256个桶
      const shift = byte << 3; // 等价于 byte * 8

      for (const num of nums) {
        view.setUint32(0, num);
        const byteVal = view.getUint8(byte);
        buckets[byteVal].push(num);
      }
      nums = [].concat(...buckets);
    }
    return nums;
  }

  return [
    ...sort(negatives)
      .reverse()
      .map((n) => -n), // 负数部分倒序
    ...sort(positives),
  ];
}
